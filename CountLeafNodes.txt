// Code to Find Number Of Leaf Nodes in A Binary Tree :-

// Method 1 :-
int CountNodes(TreeNode* root){
    if(root->left==NULL && root->right==NULL)
        return 1;

    return CountNodes(root->left) + CountNodes(root->right);
    
}

//Method 2 - using Traversal :-
int CountNodes(TreeNode* root){
    int count=0;
    traversal(root,&count);
    return count;
}
void Traversal(TreeNode* root,int* ct){
    if(root->left==NULL && root->right==NULL){
     *ct = *ct + 1;
     return;
    }

    Traversal(root->left,ct);
    Traversal(root->right,ct);
}
